import { Meta, ColorPalette, ColorItem } from '@storybook/addon-docs/blocks';

import { colors } from '../../src/theme/overwrites/foundations/colors.ts';

<Meta title="Design System / Colors" />

export const printAllVariantsByColorName = (object, colorName) => {
  const selectedColor = object[colorName];
  return Object.keys(selectedColor).map((value) => (
    <ColorItem title={`${colorName}.${value}`} subtitle={selectedColor[value]} colors={[selectedColor[value]]} />
  ));
};

export const printAllColorsInOneItem = (color, title) => {
  return <ColorItem title={title} subtitle="" colors={Object.values(colors[color])} />;
};

# Brand

<ColorPalette>
  {printAllColorsInOneItem('brand')}
  {printAllVariantsByColorName(colors, 'brand')}
</ColorPalette>
