import { Meta } from '@storybook/addon-docs';

import { IconsTable } from '../components/IconsTable';
import { IconCell } from '../components/IconCell';
import { ThemeProvider } from '../../src/theme';
import * as icons from '../../src/icons';

<Meta title="Design System / Icons" />

# Outlined icons

<ThemeProvider>
  <IconsTable>
    {Object.values(icons)
      .filter(({ displayName }) => displayName.includes('Outlined'))
      .map((icon) => (
        <IconCell key={icon.displayName} icon={icon} />
      ))}
  </IconsTable>
</ThemeProvider>
